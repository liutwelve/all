// 热区图片
		let hotZoneImgSwiper = new Swiper('.hotZiPart', {
			pagination : '.swiper-pagination',
			loop: false, //是否可循环
			observer: true, 
			observeParents: true, //监测Swiper 的祖/父元素 
			autoplayDisableOnInteraction: false,
		});
		$('.hotZiPart .swiper-slide').each(function(i){
			let json = JSON.parse($(this).attr("json"))
			console.log(json,'json')
			let area = ''
			json.map((v) => {
				const [x,y,w,h] = v.areaMapInfo.split(',')
				// console.log(v.areaMapInfo.split(','),x,y,w,h)
				area += `<area shape="rect" coords="${compatible(x)},${compatible(y)},${compatible(w)},${compatible(h)}" alt="${v.areaTitle}" href="${v.areaLink}">`
			})
			$(this).find('img').attr("usemap","#map"+i).after(`<map name="map${i}">${area}</map>`)
		})
		function compatible(num) {
			if(parseFloat(num) == 0){
				return 0
			}
			return (parseFloat(num)/375)*window.innerWidth
		}
